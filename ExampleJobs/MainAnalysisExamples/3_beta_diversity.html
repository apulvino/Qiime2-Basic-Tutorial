#!/bin/bash

  #SBATCH --account=
  #SBATCH --partition=
  #SBATCH --job-name=qiime_summaryinfo
  #SBATCH --output=qiime_summaryinfo.out
  #SBATCH --error=qiime_summaryinfo.err
  #SBATCH --time=00:00:00
  #SBATCH --mem=G
  #SBATCH --ntasks-per-node=
  #SBATCH --cpus-per-task=
  #SBATCH --mail-user=firstname.lastname@northwestern.edu
  #SBATCH --mail-type=END

  module load qiime2/2022.8
  
  qiime diversity core-metrics-phylogenetic --i-phylogeny /path/to/your/output/rooted_tree.qza --i-table /path/to/your/output/table_dada2_nomito_nochloro.qza \
  --p-sampling-depth  --m-metadata-file /path/to/your/.txt --output-dir /path/to/your/output/core-metrics--results

  qiime tools extract --input-path /path/to/your/output/core-metrics--results/unweighted_unifrac_distance_matrix.qza \
  --output-path /path/to/your/output/core-metrics--results/

  qiime tools extract --input-path /path/to/your/output/core-metrics--results/weighted_unifrac_distance_matrix.qza \
  --output-path /path/to/your/output/core-metrics--results/

  qiime emperor plot --i-pcoa  /path/to/your/output/core-metrics--results/weighted_unifrac_pcoa_results.qza \
  --m-metadata-file /path/to/your/.txt --o-visualization /path/to/your/output/emperor_weighted_unifrac_pcoa_results.qzv
